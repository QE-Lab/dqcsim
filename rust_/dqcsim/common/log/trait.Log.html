use dqcsim::{
    common::log::{Log, Loglevel, LogRecord}
};

struct SimpleLogger {}

impl Log for SimpleLogger {
    fn name(&self) -> &str {
        "simple_logger"
    }

    fn enabled(&self, level: Loglevel) -> bool {
        // The SimpleLogger is always enabled.
        true
    }

    fn log(&self, record: &LogRecord) {
        // The SimpleLogger logs to Standard Output.
        println!("{}", record);
    }
}
